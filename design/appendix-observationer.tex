	{Procyon} % Producent
	{Måle størrelsen af pupillen under forskellige lysstyrker} % Formål
	{Undersøgelse 50} % Placering
	{Lene Pedersen (læge)} % Bruger
\begin{itemize}

\item Patient oprettes manuelt i PupilFit.
\item Den nyoprettede patient genfindes og vælges.
\item Der trykkes på knappen '1'.

	måling aflæses på skærmen og indtastes i PC-Praxis.

\section{Fundus Kamera}
\apparat{Fundus Kamera} % Apparat
	{Canon} % Producent
	{Tager billeder af retina} % Formål
	{Stue 230} % Placering
	{Vivi Frisk (sygeplejerske)} % Bruger
\begin{itemize}
\item Patienten ankommer og skrives på en liste (papir).
\item Patienten sættes til rette.
\item \textit{A}. Der stilles skarpt.
\item \textit{B}. Der tages billede.
\item \textit{C}. Billedet kontrolleres på kameraet.
\item Punkterne \textit{A}, \textit{B} og \textit{C} gentages til alle
	de ønskede motiver er produceret.

\item Billederne hentes fra kameraet ved tryk på 'Hent'.
\item Billederne markeres med øje og position.
\item Der trykkes 'Gem'.
\item Nogle af billederne blev konstateret for mørke og \textit{A},
	\textit{B}, \textit{C} samt indtastning af CPR-nummeret og sortering gentages
	for at producere nye.
\end{itemize}
